дҪҝз”Ёзҡ„дҫқиө–жҳҜпјҡS7NetPlusпјҢзӣҙжҺҘеҺ»NuGetдёӢиҪҪе®үиЈ…е°ұиЎҢгҖӮ

дҪҝз”ЁS7еҚҸи®®пјҢжҲ‘д»¬дё»иҰҒжҳҜдёәдәҶе’ҢиҘҝй—ЁеӯҗPLCиҝӣиЎҢйҖҡдҝЎпјҢд№ҹе°ұжҳҜиҜ»е’ҢеҶҷпјӣиҜ»еҶҷзҡ„еүҚжҸҗжҳҜжҲ‘д»¬йңҖиҰҒе°ҶPLCзҡ„еҸҳйҮҸеӯҳеӮЁеҷЁең°еқҖиҪ¬жҚўдёәS7еҚҸи®®еҸҜд»ҘжҺҘеҸ—зҡ„ж•°жҚ®еқ—ең°еқҖпјӣз„¶еҗҺиҜ»пјҢжҲ‘д»¬йңҖиҰҒиҪ¬жҚўдёәз”ЁжҲ·еҸҜд»ҘзңӢжҮӮзҡ„ж•°жҚ®зұ»еһӢгҖӮ

PLCзҡ„еҸҳйҮҸеӯҳеӮЁеҷЁең°еқҖиҪ¬жҚўдёәж•°жҚ®еқ—ең°еқҖ

        // ең°еқҖиҪ¬жҚўж–№жі•
        private string ConvertVAddress(string address)
        {
            // еҢ№й…Қ V еҢәдҪҚең°еқҖзҡ„жӯЈеҲҷиЎЁиҫҫејҸ (еҰӮ V102.0, V102.1)
            var matchBit = Regex.Match(address, @"^V(\d+)\.(\d+)$");
            if (matchBit.Success)
            {
                int byteOffset = int.Parse(matchBit.Groups[1].Value);
                int bitOffset = int.Parse(matchBit.Groups[2].Value);
                return $"DB1.DBX{byteOffset}.{bitOffset}";
            }

            // еҢ№й…Қ VB еҢәеӯ—иҠӮең°еқҖзҡ„жӯЈеҲҷиЎЁиҫҫејҸ (еҰӮ VB100)
            var matchByte = Regex.Match(address, @"^VB(\d+)$");
            if (matchByte.Success)
            {
                int byteOffset = int.Parse(matchByte.Groups[1].Value);
                return $"DB1.DBB{byteOffset}";  // дҪҝз”Ё DBB иЎЁзӨәеӯ—иҠӮи®ҝй—®
            }

            // еҢ№й…Қ V еҢәең°еқҖзҡ„жӯЈеҲҷиЎЁиҫҫејҸ (дёҚеёҰдҪҚзҡ„ең°еқҖ)
            var match = Regex.Match(address, @"^V(\d+)$");
            if (match.Success)
            {
                int byteOffset = int.Parse(match.Groups[1].Value);
                return $"DB1.DBX{byteOffset}.0"; // й»ҳи®Өи®ҝй—®з¬¬0дҪҚ
            }

            // еҢ№й…Қ VW еҢәең°еқҖзҡ„жӯЈеҲҷиЎЁиҫҫејҸ
            match = Regex.Match(address, @"^VW(\d+)$");
            if (match.Success)
            {
                int wordOffset = int.Parse(match.Groups[1].Value);
                return $"DB1.DBW{wordOffset}";
            }

            // еҢ№й…Қ VD еҢәең°еқҖзҡ„жӯЈеҲҷиЎЁиҫҫејҸ
            match = Regex.Match(address, @"^VD(\d+)$");
            if (match.Success)
            {
                int dwordOffset = int.Parse(match.Groups[1].Value);
                return $"DB1.DBD{dwordOffset}";
            }

            // еҰӮжһңдёҚжҳҜ V еҢәең°еқҖпјҢзӣҙжҺҘиҝ”еӣһеҺҹең°еқҖ
            return address;
        }
