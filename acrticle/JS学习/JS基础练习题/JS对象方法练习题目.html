let person = {firstName: "李", lastName: "四"};

题目 6

题目 7

题目 8

let data = {a: 1, b: 2, c: 3};
// 使用Object.keys()和forEach遍历对象的所有属性

题目 9

题目 10

// 使用hasOwnProperty()检查产品是否有价格属性，然后转换为字符串

题目 12

题目 17

题目 18

题目 19

题目 20

题目 22

题目 23

    return this.firstName + this.lastName;
}};
