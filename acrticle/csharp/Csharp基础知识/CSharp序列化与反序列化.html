C# еәҸеҲ—еҢ–дёҺеҸҚеәҸеҲ—еҢ–

еәҸеҲ—еҢ–е’ҢеҸҚеәҸеҲ—еҢ–жҳҜзј–зЁӢдёӯеёёи§Ғзҡ„жҰӮеҝөпјҡ

  • еәҸеҲ—еҢ–(Serialization): е°ҶеҜ№иұЎиҪ¬жҚўдёәеҸҜд»ҘеӯҳеӮЁжҲ–дј иҫ“зҡ„ж јејҸпјҲйҖҡеёёжҳҜеӯ—з¬ҰдёІпјү
  • еҸҚеәҸеҲ—еҢ–(Deserialization): е°ҶеәҸеҲ—еҢ–еҗҺзҡ„еӯ—з¬ҰдёІиҪ¬жҚўеӣһеҺҹе§ӢеҜ№иұЎ(C# жҳҜејәзұ»еһӢиҜӯиЁҖпјҢеҝ…йЎ»е…Ҳе®ҡд№үзұ»пјҢз„¶еҗҺжүҚиғҪеҲӣе»әеҜ№иұЎпјҲе®һдҫӢпјү)

еңЁ C# дёӯпјҢдё»иҰҒйҖҡиҝҮ Newtonsoft.Json еә“е®һзҺ°гҖӮ

ж ёеҝғж–№жі•

1. JsonConvert.SerializeObject() - еәҸеҲ—еҢ–

// C# жҳҜејәзұ»еһӢиҜӯиЁҖпјҢеҝ…йЎ»е…Ҳе®ҡд№үзұ»пјҢз„¶еҗҺжүҚиғҪеҲӣе»әеҜ№иұЎпјҲе®һдҫӢпјү
public class Person
{
    public string Name { get; set; }
    public int Age { get; set; }
    public string[] Hobbies { get; set; }
}

// еҲӣе»әеҜ№иұЎе№¶еәҸеҲ—еҢ–
// var жҳҜзұ»еһӢжҺЁж–ӯпјҢзј–иҜ‘еҷЁзҹҘйҒ“ person зҡ„зұ»еһӢжҳҜ Person
var person = new Person
{
    Name = "еј дёү",
    Age = 30,
    Hobbies = new[] { "йҳ…иҜ»", "жёёжіі" }
};

string jsonString = JsonConvert.SerializeObject(person);
Console.WriteLine(jsonString);
// иҫ“еҮә: {"Name":"еј дёү","Age":30,"Hobbies":["йҳ…иҜ»","жёёжіі"]}

2. JsonConvert.DeserializeObject() - еҸҚеәҸеҲ—еҢ–

T иЎЁзӨәжіӣеһӢзұ»еһӢеҸӮж•°пјҢжҢҮе®ҡиҰҒе°Ҷ JSON еҸҚеәҸеҲ—еҢ–дёәд»Җд№Ҳзұ»еһӢзҡ„еҜ№иұЎгҖӮ

string jsonString = "{\"Name\":\"еј дёү\",\"Age\":30}";
Person person = JsonConvert.DeserializeObject<Person>(jsonString);
Console.WriteLine(person.Name); // иҫ“еҮә: еј дёү

3. JObject.Parse() - еҠЁжҖҒи§Јжһҗ

еҪ“дёҚйңҖиҰҒе®Ңж•ҙеҜ№иұЎж—¶пјҢеҸҜд»ҘдҪҝз”ЁеҠЁжҖҒи§Јжһҗпјҡ

using Newtonsoft.Json.Linq;

string jsonString = "{\"Name\":\"еј дёү\",\"Age\":30}";
JObject jObject = JObject.Parse(jsonString);
Console.WriteLine($"е§“еҗҚ: {jObject["Name"]}"); // иҫ“еҮә: е§“еҗҚ: еј дёү

C# дёӯзҡ„еәҸеҲ—еҢ–е’ҢеҸҚеәҸеҲ—еҢ–дё»иҰҒйҖҡиҝҮпјҡ

  • JsonConvert.SerializeObject() - еәҸеҲ—еҢ–
  • JsonConvert.DeserializeObject<T>() - ејәзұ»еһӢеҸҚеәҸеҲ—еҢ–
  • JObject.Parse() - еҠЁжҖҒи§Јжһҗ

3. JObject.Parse() - еҠЁжҖҒи§Јжһҗзҡ„иҜҰз»ҶзӨәдҫӢ

using Newtonsoft.Json.Linq;

string json = @"{
  ""Company"": ""и…ҫи®Ҝ"",
  ""CTO"": {
    ""Name"": ""еј е°Ҹйҫҷ"",
    ""Age"": 54,
    ""Dept"": ""еҫ®дҝЎдәӢдёҡзҫӨ""
  }
}";

// 1. и§Јжһҗж•ҙдёӘ JSON дёә JObject
JObject root = JObject.Parse(json);

// 2. д»Һ JObject еҸ–еҖј вҶ’ еҫ—еҲ°зҡ„йғҪжҳҜ JTokenпјҒ
JToken companyToken = root["Company"]; // е®һйҷ…жҳҜ JValueпјҲеӯ—з¬ҰдёІпјү
JToken ctoToken = root["CTO"]; // root["CTO"]е®һйҷ…жҳҜ JObject(Newtonsoft.Jsonзҡ„jsonзұ»еһӢ)пјҲеӣ дёә CTO жҳҜ jsonпјү

// 3. еҰӮжһңеҸӘжҳҜиҜ»еӯ—з¬ҰдёІпјҢзӣҙжҺҘиҪ¬ stringпјҲдёҚйңҖиҰҒ JObjectпјҒпјү
string company = (string)companyToken; // вң… жӯЈзЎ®пјҡз”Ё (string)
Console.WriteLine($"е…¬еҸё: {company}"); // иҫ“еҮә: е…¬еҸё: и…ҫи®Ҝ

// 4. дҪҶеҰӮжһңжғіжҠҠ CTO еҪ“дҪңеҜ№иұЎжқҘж“ҚдҪңпјҲжҜ”еҰӮиҜ»е®ғзҡ„еұһжҖ§гҖҒеҠ еӯ—ж®өпјүпјҢе°ұеҝ…йЎ»жҠҠ JToken иҪ¬жҲҗ JObjectпјҒ
// еҪ“иҝҷдёӘ JToken д»ЈиЎЁдёҖдёӘ JSON еҜ№иұЎпјҲ{}пјүпјҢиҖҢдҪ йңҖиҰҒжҠҠе®ғеҪ“дҪң JObject жқҘж“ҚдҪңж—¶пјҢжүҚйңҖиҰҒз”Ё (JObject) иҪ¬жҚўгҖӮ
JObject ctoObject = (JObject)ctoToken; // рҹ‘Ҳ иҝҷйҮҢз”ЁеҲ°дәҶ (JObject)пјҒ

// зҺ°еңЁеҸҜд»ҘеғҸж“ҚдҪңжҷ®йҖҡ JObject дёҖж ·ж“ҚдҪңе®ғпјҡ
string name = (string)ctoObject["Name"];
int age = (int)ctoObject["Age"];
Console.WriteLine($"{name}, {age}еІҒ"); // иҫ“еҮә: еј е°Ҹйҫҷ, 54еІҒ

// д№ҹеҸҜд»Ҙ
JObject Cto_Object = (JObject)root["CTO"];

// иҝҳеҸҜд»Ҙж·»еҠ ж–°еұһжҖ§пјҲеҸӘжңү JObject ж”ҜжҢҒпјүпјҡ
ctoObject["Email"] = "zhangxiaolong@tencent.com";

// йӘҢиҜҒжҳҜеҗҰжҲҗеҠҹпјҡJTokenе’ҢJObjectйғҪеҸҜд»ҘзӣҙжҺҘжү“еҚ°пјҢдҪҶжҳҜеҪ“иҝҷдёӘ JToken д»ЈиЎЁдёҖдёӘ JSON еҜ№иұЎпјҲ{}пјүпјҢе№¶дё”дҪ йңҖиҰҒжҠҠе®ғеҪ“дҪң еҜ№иұЎ жқҘж“ҚдҪңж—¶пјҢжүҚйңҖиҰҒз”Ё (JObject) иҪ¬жҚўгҖӮ
Console.WriteLine(root["CTO"]["Email"]); // иҫ“еҮә: zhangxiaolong@tencent.com

JObject е’Ң JToken зҡ„еҢәеҲ«

  • JToken: жҳҜ Newtonsoft.Json еә“дёӯжүҖжңү JSON еҖјзҡ„еҹәзұ»пјҢеҸҜд»ҘиЎЁзӨәд»»дҪ• JSON ж•°жҚ®зұ»еһӢпјҲеҜ№иұЎгҖҒж•°з»„гҖҒеӯ—з¬ҰдёІгҖҒж•°еӯ—зӯүпјү
  • JObject: з»§жүҝиҮӘ JTokenпјҢдё“й—Ёз”ЁдәҺиЎЁзӨә JSON еҜ№иұЎпјҲй”®еҖјеҜ№йӣҶеҗҲпјү

дҪҝз”ЁеңәжҷҜ

  1. еҸӘйңҖиҰҒиҜ»еҸ–з®ҖеҚ•еҖјпјҡзӣҙжҺҘдҪҝз”Ё (string) жҲ– (int) зӯүзұ»еһӢиҪ¬жҚў
  2. йңҖиҰҒж“ҚдҪң JSON еҜ№иұЎпјҡе°Ҷ JToken иҪ¬жҚўдёә JObject еҗҺиҝӣиЎҢж“ҚдҪң
  3. йңҖиҰҒж·»еҠ /дҝ®ж”№еұһжҖ§пјҡеҝ…йЎ»дҪҝз”Ё JObjectпјҢеӣ дёәеҸӘжңү JObject ж”ҜжҢҒж·»еҠ ж–°еұһжҖ§
