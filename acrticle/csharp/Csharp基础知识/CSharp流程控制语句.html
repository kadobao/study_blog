C#жөҒзЁӢжҺ§еҲ¶иҜӯеҸҘ

XJ2025/9/5еӨ§зәҰ 4 еҲҶй’ҹC#еӯҰд№ C#еҹәзЎҖ

C# жөҒзЁӢжҺ§еҲ¶иҜӯеҸҘиҜҰи§Ј

жҰӮиҝ°

жөҒзЁӢжҺ§еҲ¶иҜӯеҸҘжҳҜзј–зЁӢиҜӯиЁҖдёӯз”ЁдәҺжҺ§еҲ¶д»Јз Ғжү§иЎҢйЎәеәҸзҡ„е…ій”®йғЁеҲҶгҖӮC# жҸҗдҫӣдәҶеӨҡз§ҚжөҒзЁӢжҺ§еҲ¶з»“жһ„пјҢз”ЁдәҺе®һзҺ°жқЎд»¶еҲӨж–ӯгҖҒеҫӘзҺҜжү§иЎҢе’ҢејӮеёёеӨ„зҗҶзӯүйҖ»иҫ‘гҖӮ

1. If-Else жқЎд»¶иҜӯеҸҘ

жҸҸиҝ°пјҡж №жҚ®жқЎд»¶зҡ„зңҹеҒҮжқҘжү§иЎҢдёҚеҗҢзҡ„д»Јз Ғеқ—гҖӮ

еҹәжң¬з”Ёжі•

int score = 85;

if (score >= 90)
{
    Console.WriteLine("дјҳз§Җ");
}
else if (score >= 60)
{
    Console.WriteLine("еҸҠж ј");
}
else
{
    Console.WriteLine("дёҚеҸҠж ј");
}

еёёз”Ёж“ҚдҪң

// еөҢеҘ— If
bool isLoggedIn = true;
bool isAdmin = false;

if (isLoggedIn)
{
    Console.WriteLine("з”ЁжҲ·е·Ізҷ»еҪ•");
    if (isAdmin)
    {
        Console.WriteLine("ж¬ўиҝҺз®ЎзҗҶе‘ҳ");
    }
    else
    {
        Console.WriteLine("ж¬ўиҝҺжҷ®йҖҡз”ЁжҲ·");
    }
}
else
{
    Console.WriteLine("иҜ·е…Ҳзҷ»еҪ•");
}

// дёүе…ғиҝҗз®—з¬Ұ (з®ҖеҢ–зүҲ If-Else)
int a = 10;
int b = 20;
int max = (a > b) ? a : b; // еҰӮжһң a > b дёәзңҹпјҢеҲҷиҝ”еӣһ aпјҢеҗҰеҲҷиҝ”еӣһ b
Console.WriteLine($"жңҖеӨ§еҖјжҳҜ: {max}"); // иҫ“еҮә "жңҖеӨ§еҖјжҳҜ: 20"

2. For еҫӘзҺҜ

жҸҸиҝ°пјҡеңЁж»Ўи¶іжҢҮе®ҡжқЎд»¶ж—¶пјҢйҮҚеӨҚжү§иЎҢдёҖж®өд»Јз ҒгҖӮйҖҡеёёз”ЁдәҺе·ІзҹҘеҫӘзҺҜж¬Ўж•°зҡ„еңәжҷҜгҖӮ

еҹәжң¬з”Ёжі•

// д»Һ 0 еҫӘзҺҜеҲ° 4пјҢе…ұ 5 ж¬Ў
for (int i = 0; i < 5; i++)
{
    Console.WriteLine($"еҪ“еүҚж•°еӯ—жҳҜ: {i}");
}

еёёз”Ёж“ҚдҪң

// йҒҚеҺҶж•°з»„
int[] numbers = { 10, 20, 30, 40, 50 };
for (int i = 0; i < numbers.Length; i++)
{
    Console.WriteLine($"зҙўеј• {i} зҡ„еҖјдёә: {numbers[i]}");
}

// еөҢеҘ— For еҫӘзҺҜ (жү“еҚ°д№қд№қд№ҳжі•иЎЁ)
for (int i = 1; i <= 9; i++)
{
    for (int j = 1; j <= i; j++)
    {
        Console.Write($"{j} * {i} = {i * j}\t");
    }
    Console.WriteLine(); // жҚўиЎҢ
}

// break е’Ң continue
for (int i = 0; i < 10; i++)
{
    if (i == 5)
    {
        break; // еҪ“ i зӯүдәҺ 5 ж—¶пјҢз«ӢеҚіз»ҲжӯўеҫӘзҺҜ
    }
    if (i % 2 == 0)
    {
        continue; // еҪ“ i жҳҜеҒ¶ж•°ж—¶пјҢи·іиҝҮжң¬ж¬ЎеҫӘзҺҜзҡ„еү©дҪҷйғЁеҲҶпјҢиҝӣе…ҘдёӢдёҖж¬ЎеҫӘзҺҜ
    }
    Console.WriteLine(i); // еҸӘдјҡжү“еҚ° 1, 3
}

3. Foreach еҫӘзҺҜ

жҸҸиҝ°пјҡз”ЁдәҺйҒҚеҺҶйӣҶеҗҲпјҲеҰӮж•°з»„гҖҒеҲ—иЎЁгҖҒеӯ—е…ёзӯүпјүдёӯзҡ„жҜҸдёӘе…ғзҙ гҖӮд»Јз Ғжӣҙз®ҖжҙҒпјҢеҸҜиҜ»жҖ§жӣҙй«ҳгҖӮ

foreach жӢҝеҲ°зҡ„еҫӘзҺҜеҸҳйҮҸжҳҜеҜ№иұЎзҡ„еј•з”Ёпјӣдҝ®ж”№еҫӘзҺҜеҸҳйҮҸйҮҢйқўзҡ„еұһжҖ§жҲ–еӯ—ж®өпјҢеҺҹеҲ—иЎЁгҖҒеӯ—е…ёйҮҢзҡ„е…ғзҙ дјҡеҗҢжӯҘеҸҳеҢ–гҖӮ

4. While еҫӘзҺҜ

жҸҸиҝ°пјҡеҸӘиҰҒжҢҮе®ҡжқЎд»¶дёәзңҹпјҢе°ұйҮҚеӨҚжү§иЎҢд»Јз Ғеқ—гҖӮйҖӮз”ЁдәҺеҫӘзҺҜж¬Ўж•°жңӘзҹҘзҡ„еңәжҷҜгҖӮ

еҹәжң¬з”Ёжі•

int count = 0;
while (count < 5)
{
    Console.WriteLine($"еҪ“еүҚи®Ўж•°: {count}");
    count++;
}

5. Try-Catch ејӮеёёеӨ„зҗҶ

жҸҸиҝ°пјҡз”ЁдәҺжҚ•иҺ·е’ҢеӨ„зҗҶд»Јз ҒеңЁиҝҗиЎҢж—¶еҸҜиғҪеҸ‘з”ҹзҡ„й”ҷиҜҜпјҲејӮеёёпјүпјҢд»ҘйҳІжӯўзЁӢеәҸеҙ©жәғгҖӮ

еҹәжң¬з”Ёжі•

try
{
    // еҸҜиғҪдјҡжҠӣеҮәејӮеёёзҡ„д»Јз Ғ
    int zero = 0;
    int result = 10 / zero; // иҝҷе°Ҷеј•еҸ‘ DivideByZeroException
    Console.WriteLine(result);
}
catch (DivideByZeroException ex)
{
    // жҚ•иҺ·зү№е®ҡзұ»еһӢзҡ„ејӮеёё
    Console.WriteLine("й”ҷиҜҜпјҡдёҚиғҪйҷӨд»Ҙйӣ¶гҖӮ");
    Console.WriteLine($"ејӮеёёдҝЎжҒҜ: {ex.Message}");
}
catch (Exception ex)
{
    // жҚ•иҺ·жүҖжңүе…¶д»–зұ»еһӢзҡ„ејӮеёё (йҖҡеёёж”ҫеңЁжңҖеҗҺ)
    Console.WriteLine("еҸ‘з”ҹдәҶдёҖдёӘжңӘзҹҘй”ҷиҜҜгҖӮ");
    Console.WriteLine($"ејӮеёёдҝЎжҒҜ: {ex.Message}");
}

finally е’Ң throw

жҸҸиҝ°пјҡfinally еқ—дёӯзҡ„д»Јз Ғж— и®әжҳҜеҗҰеҸ‘з”ҹејӮеёёйғҪдјҡжү§иЎҢпјҢйҖҡеёёз”ЁдәҺиө„жәҗйҮҠж”ҫгҖӮthrow з”ЁдәҺжүӢеҠЁжҠӣеҮәејӮеёёгҖӮ

FileStream file = null;
try
{
    file = new FileStream("path/to/file.txt", FileMode.Open);
    // ... ж–Үд»¶ж“ҚдҪң ...

    // жүӢеҠЁжҠӣеҮәејӮеёё
    throw new InvalidOperationException("ж“ҚдҪңж— ж•Ҳ");
}
catch (FileNotFoundException ex)
{
    Console.WriteLine("ж–Үд»¶жңӘжүҫеҲ°гҖӮ");
}
catch (Exception ex)
{
    Console.WriteLine($"жҚ•иҺ·еҲ°ејӮеёё: {ex.Message}");
    // йҮҚж–°жҠӣеҮәејӮеёёпјҢи®©дёҠеұӮи°ғз”ЁиҖ…еӨ„зҗҶ
    throw;
}
finally
{
    // зЎ®дҝқж–Үд»¶еҸҘжҹ„иў«е…ій—ӯпјҢеҚідҪҝеҸ‘з”ҹејӮеёё
    if (file != null)
    {
        file.Close();
        Console.WriteLine("ж–Үд»¶е·Іе…ій—ӯгҖӮ");
    }
}

жҖ»з»“

йҖүжӢ©еҗҲйҖӮзҡ„еҫӘзҺҜ

  • forпјҡеҪ“дҪ жҳҺзЎ®зҹҘйҒ“йңҖиҰҒеҫӘзҺҜеӨҡе°‘ж¬Ўж—¶пјҲдҫӢеҰӮпјҢйҒҚеҺҶж•°з»„зҡ„зҙўеј•пјүгҖӮ
  • foreachпјҡеҪ“дҪ йңҖиҰҒйҒҚеҺҶдёҖдёӘйӣҶеҗҲ(ListгҖҒArrayгҖҒDictionaryзӯү)дёӯзҡ„жүҖжңүе…ғзҙ пјҢиҖҢдёҚйңҖиҰҒе…іеҝғзҙўеј•ж—¶гҖӮд»Јз ҒжңҖз®ҖжҙҒгҖӮ
  • whileпјҡеҪ“еҫӘзҺҜзҡ„ж¬Ўж•°дёҚзЎ®е®ҡпјҢеҸ–еҶідәҺжҹҗдёӘжқЎд»¶жҳҜеҗҰж»Ўи¶іж—¶гҖӮ
  • do-whileпјҡдёҺ while зұ»дјјпјҢдҪҶйңҖиҰҒзЎ®дҝқеҫӘзҺҜдҪ“иҮіе°‘жү§иЎҢдёҖж¬ЎгҖӮ
