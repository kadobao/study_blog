C#ж•°жҚ®зұ»еһӢжү“еҚ°ж–№жі•

XJ2025/9/5еӨ§зәҰ 4 еҲҶй’ҹC#еӯҰд№ C#еҹәзЎҖ

C# ж•°жҚ®зұ»еһӢжү“еҚ°ж–№жі•иҜҰи§Ј

жҰӮиҝ°

еңЁ C# дёӯпјҢдәҶи§ЈеҰӮдҪ•иҺ·еҸ–е’Ңжү“еҚ°ж•°жҚ®зұ»еһӢжҳҜиҝӣиЎҢзұ»еһӢжЈҖжҹҘгҖҒи°ғиҜ•е’ҢеҸҚе°„зј–зЁӢзҡ„еҹәзЎҖгҖӮC# жҸҗдҫӣдәҶеӨҡз§ҚеҶ…зҪ®жңәеҲ¶жқҘиҺ·еҸ–еҸҳйҮҸзҡ„иҝҗиЎҢж—¶зұ»еһӢжҲ–е·ІзҹҘзұ»еһӢзҡ„зј–иҜ‘ж—¶дҝЎжҒҜпјҢе№¶д»ҘдёҚеҗҢеҪўејҸпјҲеҰӮе®Ңж•ҙеҗҚз§°гҖҒз®ҖеҚ•еҗҚз§°пјүиҝӣиЎҢеұ•зӨәгҖӮ

1. дҪҝз”Ё GetType() ж–№жі•

жҸҸиҝ°пјҡGetType() жҳҜ object зұ»зҡ„ж–№жі•пјҢз”ЁдәҺиҺ·еҸ–еҜ№иұЎзҡ„иҝҗиЎҢж—¶зұ»еһӢпјҢиҝ”еӣһдёҖдёӘ Type еҜ№иұЎгҖӮжӯӨж–№жі•йҖӮз”ЁдәҺжүҖжңүе®һдҫӢпјҢеӣ дёәе®ғз»§жүҝиҮӘ objectгҖӮ

еҹәжң¬з”Ёжі•

int number = 42;
Console.WriteLine(number.GetType()); // иҫ“еҮә: System.Int32

string text = "Hello";
Console.WriteLine(text.GetType()); // иҫ“еҮә: System.String

bool flag = true;
Console.WriteLine(flag.GetType()); // иҫ“еҮә: System.Boolean

еёёз”Ёж“ҚдҪң

// иҺ·еҸ–з©әеј•з”Ёзҡ„зұ»еһӢдјҡеҜјиҮҙ NullReferenceExceptionпјҢйңҖиҰҒиҝӣиЎҢжЈҖжҹҘ
string nullText = null;
if (nullText != null)
{
    Console.WriteLine(nullText.GetType());
}
else
{
    Console.WriteLine("nullText дёә nullпјҢж— жі•иҺ·еҸ–иҝҗиЎҢж—¶зұ»еһӢгҖӮ");
}

object obj = 123.45m; // decimal зұ»еһӢ
Console.WriteLine(obj.GetType()); // иҫ“еҮә: System.Decimal

2. дҪҝз”Ё typeof иҝҗз®—з¬Ұ

жҸҸиҝ°пјҡtypeof иҝҗз®—з¬Ұз”ЁдәҺеңЁзј–иҜ‘ж—¶иҺ·еҸ–е·ІзҹҘзұ»еһӢзҡ„ Type еҜ№иұЎгҖӮе®ғдёҚйҖӮз”ЁдәҺеҸҳйҮҸе®һдҫӢпјҢеҸӘйҖӮз”ЁдәҺзұ»еһӢеҗҚз§°гҖӮ

еҹәжң¬з”Ёжі•

Console.WriteLine(typeof(int));     // иҫ“еҮә: System.Int32
Console.WriteLine(typeof(string));  // иҫ“еҮә: System.String
Console.WriteLine(typeof(bool));    // иҫ“еҮә: System.Boolean

еёёз”Ёж“ҚдҪң

// жҜ”иҫғзұ»еһӢ
Type intType = typeof(int);
Type numberType = number.GetType(); // еҒҮи®ҫ number жҳҜдёҖдёӘ int еҸҳйҮҸ
Console.WriteLine(intType == numberType); // иҫ“еҮә: True

// typeof йҖӮз”ЁдәҺд»»дҪ•еҶ…зҪ®жҲ–иҮӘе®ҡд№үзұ»еһӢпјҢеҢ…жӢ¬жіӣеһӢе®ҡд№ү
Console.WriteLine(typeof(System.Collections.Generic.List<>)); // иҫ“еҮә: System.Collections.Generic.List`1
Console.WriteLine(typeof(Console)); // иҫ“еҮә: System.Console

3. дҪҝз”Ё GetType().Name иҺ·еҸ–з®ҖеҚ•еҗҚз§°

жҸҸиҝ°пјҡType еҜ№иұЎзҡ„ Name еұһжҖ§иҝ”еӣһзұ»еһӢзҡ„з®ҖеҚ•еҗҚз§°пјҢдёҚеҢ…еҗ«е‘ҪеҗҚз©әй—ҙдҝЎжҒҜгҖӮеҜ№дәҺжіӣеһӢзұ»еһӢпјҢеҗҚз§°еҗҺдјҡи·ҹйҡҸеҸҚеј•еҸ·е’Ңзұ»еһӢеҸӮж•°зҡ„ж•°йҮҸгҖӮ

еҹәжң¬з”Ёжі•

int[] numbers = {1, 2, 3};
Console.WriteLine(numbers.GetType().Name); // иҫ“еҮә: Int32[]

System.Collections.Generic.List<string> list = new System.Collections.Generic.List<string>();
Console.WriteLine(list.GetType().Name); // иҫ“еҮә: List`1

string text = "Hello";
Console.WriteLine(text.GetType().Name); // иҫ“еҮә: String

еёёз”Ёж“ҚдҪң

// иҮӘе®ҡд№үзұ»еһӢзҡ„з®ҖеҚ•еҗҚз§°
public class MyClass { }
MyClass myInstance = new MyClass();
Console.WriteLine(myInstance.GetType().Name); // иҫ“еҮә: MyClass

// жіӣеһӢеӯ—е…ёзҡ„з®ҖеҚ•еҗҚз§°
System.Collections.Generic.Dictionary<string, int> dict = new System.Collections.Generic.Dictionary<string, int>();
Console.WriteLine(dict.GetType().Name); // иҫ“еҮә: Dictionary`2

4. дҪҝз”Ё GetType().FullName иҺ·еҸ–е®Ңж•ҙеҗҚз§°

жҸҸиҝ°пјҡType еҜ№иұЎзҡ„ FullName еұһжҖ§иҝ”еӣһзұ»еһӢзҡ„е®Ңж•ҙеҗҚз§°пјҢеҢ…жӢ¬е…¶е‘ҪеҗҚз©әй—ҙгҖӮеҜ№дәҺжіӣеһӢзұ»еһӢпјҢе®ғдјҡеҢ…еҗ«е®Ңж•ҙзҡ„жіӣеһӢеҸӮж•°зұ»еһӢдҝЎжҒҜгҖӮ

еҹәжң¬з”Ёжі•

int number = 42;
Console.WriteLine(number.GetType().FullName); // иҫ“еҮә: System.Int32

string text = "Hello";
Console.WriteLine(text.GetType().FullName); // иҫ“еҮә: System.String

еёёз”Ёж“ҚдҪң

// иҮӘе®ҡд№үзұ»еһӢзҡ„е®Ңж•ҙеҗҚз§°
// еҒҮи®ҫ MyClass еңЁ MyNamespace дёӯе®ҡд№ү
// namespace MyNamespace { public class MyClass { } }
// MyNamespace.MyClass myInstance = new MyNamespace.MyClass();
// Console.WriteLine(myInstance.GetType().FullName); // иҫ“еҮә: MyNamespace.MyClass

// жіӣеһӢеҲ—иЎЁзҡ„е®Ңж•ҙеҗҚз§°
System.Collections.Generic.List<int> intList = new System.Collections.Generic.List<int>();
Console.WriteLine(intList.GetType().FullName); // иҫ“еҮә: System.Collections.Generic.List`1[[System.Int32, System.Private.CoreLib, Version=...]]

5. еӨ„зҗҶжіӣеһӢзұ»еһӢ

жҸҸиҝ°пјҡC# дёӯзҡ„жіӣеһӢзұ»еһӢеңЁиҺ·еҸ–е…¶еҗҚз§°ж—¶жңүзү№ж®Ҡзҡ„иЎЁзӨәгҖӮName еұһжҖ§дјҡеҢ…еҗ«еҸҚеј•еҸ· () е’Ңзұ»еһӢеҸӮж•°зҡ„ж•°йҮҸгҖӮGetGenericArguments() ж–№жі•еҸҜд»ҘиҺ·еҸ–жіӣеһӢеҸӮж•°зҡ„ Type ж•°з»„гҖӮ

еҹәжң¬з”Ёжі•

System.Collections.Generic.List<int> intList = new System.Collections.Generic.List<int>();
System.Collections.Generic.Dictionary<string, int> dict = new System.Collections.Generic.Dictionary<string, int>();

Console.WriteLine(intList.GetType().Name); // иҫ“еҮә: List`1
Console.WriteLine(dict.GetType().Name);    // иҫ“еҮә: Dictionary`2

еёёз”Ёж“ҚдҪң

// иҺ·еҸ–жіӣеһӢеҸӮж•°дҝЎжҒҜ
Type[] genericArgs = dict.GetType().GetGenericArguments();
foreach(Type arg in genericArgs)
{
    Console.WriteLine($"жіӣеһӢеҸӮж•°еҗҚз§°: {arg.Name}"); // иҫ“еҮә: String е’Ң Int32
    Console.WriteLine($"жіӣеһӢеҸӮж•°е®Ңж•ҙеҗҚз§°: {arg.FullName}"); // иҫ“еҮә: System.String е’Ң System.Int32
}

// жЈҖжҹҘжҳҜеҗҰжҳҜжіӣеһӢзұ»еһӢ
Console.WriteLine(dict.GetType().IsGenericType); // иҫ“еҮә: True
Console.WriteLine(typeof(int).IsGenericType);    // иҫ“еҮә: False

6. иҮӘе®ҡд№үзұ»еһӢзӨәдҫӢ

жҸҸиҝ°пјҡеҜ№дәҺз”ЁжҲ·иҮӘе®ҡд№үзҡ„ classгҖҒstruct жҲ– enum зұ»еһӢпјҢдёҠиҝ°жүҖжңүж–№жі•еҗҢж ·йҖӮз”ЁпјҢд»ҘиҺ·еҸ–е…¶зұ»еһӢдҝЎжҒҜгҖӮ

еҹәжң¬з”Ёжі•

public struct Point
{
    public int X;
    public int Y;
}

public enum Status
{
    Pending,
    Approved,
    Rejected
}

// дҪҝз”ЁзӨәдҫӢ
Point p = new Point();
Status s = Status.Approved;

Console.WriteLine(p.GetType().Name);    // иҫ“еҮә: Point
Console.WriteLine(s.GetType().Name);    // иҫ“еҮә: Status
Console.WriteLine(typeof(Point).Name);  // иҫ“еҮә: Point
Console.WriteLine(typeof(Status).Name); // иҫ“еҮә: Status

еёёз”Ёж“ҚдҪң

// иҺ·еҸ–иҮӘе®ҡд№үз»“жһ„дҪ“зҡ„е®Ңж•ҙеҗҚз§°
Console.WriteLine(p.GetType().FullName); // иҫ“еҮә: YourNamespace.Point (еҸ–еҶідәҺе®һйҷ…е‘ҪеҗҚз©әй—ҙ)

// иҺ·еҸ–жһҡдёҫзҡ„еә•еұӮзұ»еһӢ
Console.WriteLine(Enum.GetUnderlyingType(typeof(Status)).Name); // иҫ“еҮә: Int32

жҖ»з»“

  • GetType() ж–№жі•пјҡз”ЁдәҺиҺ·еҸ–еҜ№иұЎзҡ„иҝҗиЎҢж—¶зұ»еһӢгҖӮеҪ“дҪ жңүдёҖдёӘеҸҳйҮҸе®һдҫӢпјҢе№¶дё”жғізҹҘйҒ“е®ғеңЁиҝҗиЎҢж—¶е®һйҷ…жҳҜд»Җд№Ҳзұ»еһӢж—¶дҪҝз”ЁгҖӮ
  • typeof иҝҗз®—з¬Ұпјҡз”ЁдәҺиҺ·еҸ–зј–иҜ‘ж—¶е·ІзҹҘзҡ„зұ»еһӢдҝЎжҒҜгҖӮеҪ“дҪ жҳҺзЎ®зҹҘйҒ“иҰҒжҹҘиҜўзҡ„зұ»еһӢеҗҚз§°ж—¶дҪҝз”ЁпјҢж— йңҖеҲӣе»әе®һдҫӢгҖӮ
  • Name еұһжҖ§пјҡиҺ·еҸ–зұ»еһӢзҡ„з®ҖеҚ•еҗҚз§°пјҢдёҚеҢ…еҗ«е‘ҪеҗҚз©әй—ҙгҖӮйҖӮз”ЁдәҺеҸӘйңҖиҰҒз®ҖжҙҒзұ»еһӢиЎЁзӨәзҡ„еңәжҷҜгҖӮ
  • FullName еұһжҖ§пјҡиҺ·еҸ–зұ»еһӢзҡ„е®Ңж•ҙеҗҚз§°пјҢеҢ…еҗ«е‘ҪеҗҚз©әй—ҙгҖӮйҖӮз”ЁдәҺйңҖиҰҒе®Ңж•ҙгҖҒе”ҜдёҖзұ»еһӢж ҮиҜҶзҡ„еңәжҷҜпјҢдҫӢеҰӮеҸҚе°„ж“ҚдҪңжҲ–еәҸеҲ—еҢ–гҖӮ
  • жіӣеһӢзұ»еһӢеӨ„зҗҶпјҡжіӣеһӢзұ»еһӢеңЁ Name дёӯдјҡеёҰ 1 жҲ– 2 зӯүеҗҺзјҖиЎЁзӨәжіӣеһӢеҸӮж•°ж•°йҮҸпјӣдҪҝз”Ё GetGenericArguments() еҸҜд»ҘиҝӣдёҖжӯҘиҺ·еҸ–жіӣеһӢеҸӮж•°зҡ„иҜҰз»Ҷзұ»еһӢгҖӮ
