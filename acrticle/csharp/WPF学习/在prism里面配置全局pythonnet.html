{
    try
    {
        // й…ҚзҪ® Python и·Ҝеҫ„
        Runtime.PythonDLL = @"C:\Users\XJ\AppData\Local\Programs\Python\Python310\python310.dll";

protected override void OnExit(ExitEventArgs e)
{
    try
    {
        // еңЁеә”з”ЁзЁӢеәҸйҖҖеҮәж—¶е…ій—ӯ Python еј•ж“Һ
        if (PythonEngine.IsInitialized)
        {
            PythonEngine.Shutdown();
        }
    }
    catch (Exception ex)
    {
        // еҝҪз•Ҙе…ій—ӯж—¶зҡ„ејӮеёё
        System.Diagnostics.Debug.WriteLine($"е…ій—ӯ Python еј•ж“Һж—¶еҮәй”ҷ: {ex.Message}");
    }
    
    base.OnExit(e);
}
